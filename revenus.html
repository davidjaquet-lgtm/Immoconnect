// Créer un lien de paiement abonnement agent (49€/mois)
app.post('/api/subscribe', async (req, res) => {
  const session = await stripe.checkout.sessions.create({
    payment_method_types: ['card'],
    mode: 'subscription',
    line_items: [{
      price_data: {
        currency: 'eur',
        product_data: { name: 'ImmoConnect Agent Pro' },
        unit_amount: 4900,        // 49€ en centimes
        recurring: { interval: 'month' }
      },
      quantity: 1
    }],
    success_url: 'https://immoconnect.fr/merci?agent=true',
    cancel_url: 'https://immoconnect.fr/agents',
  })
  res.json({ url: session.url })
})

// Vente d'un rapport DVF (19€, paiement unique)
app.post('/api/rapport-dvf', async (req, res) => {
  const { ville, codePostal } = req.body
  const session = await stripe.checkout.sessions.create({
    payment_method_types: ['card'],
    mode: 'payment',
    line_items: [{
      price_data: {
        currency: 'eur',
        product_data: {
          name: `Rapport DVF — ${ville} (${codePostal})`,
          description: 'Analyse complète des prix du marché dans votre secteur'
        },
        unit_amount: 1900   // 19€
      },
      quantity: 1
    }],
    success_url: `https://immoconnect.fr/rapport?cp=${codePostal}&paid=true`,
    cancel_url: 'https://immoconnect.fr/carte',
  })
  res.json({ url: session.url })
})

// Dans votre frontend (bouton d'achat rapport)
async function acheterRapport(ville, cp) {
  const r = await fetch('/api/rapport-dvf', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ ville, codePostal: cp })
  })
  const { url } = await r.json()
  window.location.href = url  // Redirige vers Stripe Checkout
}

function updateCredit() {
  const m = +document.getElementById('creditMontant').value
  const d = +document.getElementById('creditDuree').value
  const taux = 0.035 / 12  // 3.5% annuel
  const n = d * 12
  const mensualite = m * taux / (1 - Math.pow(1 + taux, -n))
  document.getElementById('creditMontantVal').textContent = m.toLocaleString('fr-FR') + ' €'
  document.getElementById('creditMensualite').textContent = Math.round(mensualite).toLocaleString('fr-FR') + ' €'
}

-- SQL Supabase : nouvelle table api_keys
CREATE TABLE api_keys (
  id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  created_at  TIMESTAMP DEFAULT NOW(),
  client_name TEXT NOT NULL,
  email       TEXT NOT NULL,
  api_key     TEXT UNIQUE DEFAULT encode(gen_random_bytes(32), 'hex'),
  plan        TEXT DEFAULT 'free',  -- 'free', 'pro', 'business'
  requests_this_month INTEGER DEFAULT 0,
  limit_monthly INTEGER DEFAULT 100,
  active      BOOLEAN DEFAULT TRUE
);

// Middleware de vérification de clé API
async function requireApiKey(req, res, next) {
  const key = req.headers['x-api-key']
  if (!key) return res.status(401).json({ error: 'Clé API requise' })

  const { data: client } = await supabase
    .from('api_keys')
    .select('*')
    .eq('api_key', key)
    .eq('active', true)
    .single()

  if (!client) return res.status(403).json({ error: 'Clé invalide' })
  if (client.requests_this_month >= client.limit_monthly)
    return res.status(429).json({ error: 'Quota mensuel atteint. Passez en Pro.' })

  // Incrémenter le compteur
  await supabase.from('api_keys')
    .update({ requests_this_month: client.requests_this_month + 1 })
    .eq('id', client.id)

  req.apiClient = client
  next()
}
